\paragraph{Companion app.}

\begin{appactivity}[Companion app pathway]
\begin{enumerate}
  \item \textbf{Presentation:} specify which cards or illustrations to study.
  \item \textbf{Simulation:} give a prediction, procedure, and observation prompt.
  \item \textbf{Resources:} ask the reader to open one relevant item and record what it adds.
  \item \textbf{Quiz:} state when to complete it and how to revisit errors.
\end{enumerate}
\end{appactivity}
